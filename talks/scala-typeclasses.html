Semigroup

// Scalaz
def append(a1: A, a2: => A): A

// Cats
def combine(x: A, y: A): A

Semigroup syntax

// Scalaz
def |+|(other: => F): F
def mappend(other: => F): F
def ‚äπ(other: => F): F

// Cats
def |+|(rhs: A): A
def combine(rhs: A): A

Semigroup laws

// associativity
(x |+| y) |+| z == x |+| (y |+| z)

Monoid

// Scalaz
def zero: A

// Cats
def empty: A

Monoid laws

// left identity
Monoid[A].empty |+| x == x

// right identity
x |+| Monoid[A].empty == x

Functor

def map[A, B](fa: F[A])(f: A => B): F[B]

Functor syntax

def map[B](f: A => B): F[B]

Functor laws

// identity
(x map identity) == x

// associativity
(x map (f map g)) ==
  (x map f map g)

Monad

// Scalaz
def point[A](a: => A): M[A]
def bind[A, B](a: M[A], f: A => M[B]): M[B]

// Cats
def pure[A](x: A): F[A]
def flatMap[A, B](fa: F[A])(f: A => F[B]): F[B]

Monad Laws

// left identity
(Monad[F].pure(x) flatMap {f}) == f(x)

// right identity
(m flatMap {Monad[F].pure(_)}) == m

// associativity
(m flatMap f) flatMap g ==
  m flatMap { x => f(x) flatMap {g} }
