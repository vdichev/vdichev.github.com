class TagCloud extends CometActor
  def render = {
    val messages = lookupMessages()
    <p>
        {
          ...
        }
    </p>
  }

  override def lowPriority = {
    case UserActor.MessageReceived(msg, _) =>
      messages = (msg.id.is :: messages).take(40)
      reRender(false)
  }
}
