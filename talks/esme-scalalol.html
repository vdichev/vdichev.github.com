override def getText(node: Node) = {
  // if there's no title, get description
  val title = node \ "title"
  if (title isEmpty)
    node \ "description" text
  else
    title text
}
override def getLink(node: Node) = {
  // a link is optional
  val link = node \ "link"
  if (link isEmpty)
    ""
  else
    link text
}
